**Goal:** A single `main.go` that locks the terminal, shows a minimal styled
lock screen, and unlocks via Touch ID or macOS password fallback.

**Architecture:** Single-file Go program using lipgloss for styling, CGo for
Touch ID (LocalAuthentication.framework) and PAM password auth, and x/term for
raw terminal mode.

**Tech Stack:** Go, lipgloss, golang.org/x/term, CGo (LocalAuthentication, PAM)

### Task 2: Build main.go with lock screen, Touch ID, and password auth

**Files:**

- Create: `main.go`

- [ ] **Step 1: Create main.go**

The file needs these sections:

**CGo preamble** — two C functions:

1. `authenticate_touchid()` — calls
   `LAContext evaluatePolicy:LAPolicyDeviceOwnerAuthenticationWithBiometrics`
   with reason "tlock: unlock terminal". Returns 1 on success, 0 on failure.
2. `authenticate_password(const char *pw)` — uses PAM (`pam_start`,
   `pam_authenticate`, `pam_end`) with service "login" and the current username.
   Custom PAM conversation function supplies the password. Returns 1 on success,
   0 on failure.

CGo LDFLAGS: `-framework LocalAuthentication -framework Foundation -lpam`

**Styles** — lipgloss styles using osapi palette:

- `purple = lipgloss.Color("99")`
- `teal = lipgloss.Color("#06ffa5")`
- `gray = lipgloss.Color("245")`
- Lock title style: bold, foreground purple
- Subtitle style: foreground gray
- Prompt style: foreground teal
- Error style: bold, foreground red (`lipgloss.Color("196")`)

**Terminal functions:**

- `clearScreen()` — prints `\033[2J\033[H` to clear and home cursor
- `hideCursor()` / `showCursor()` — prints `\033[?25l` / `\033[?25h`
- `getTerminalSize()` — uses `term.GetSize(int(os.Stdout.Fd()))`
- `centerText(text string, width int)` — pads text with leading spaces to center
  it

**renderLockScreen(width, height int):**

- Clears screen
- Computes vertical center
- Renders 3 lines centered:
  1. Lock title: `lockTitleStyle.Render("LOCKED")`
  2. Hostname via `os.Hostname()` in subtitle style
  3. `"Press any key to unlock"` in subtitle style
- Prints lines at vertical center of terminal

**renderAuthPrompt(width, height int, message string):**

- Clears screen
- Centers and renders message in prompt style

**renderError(width, height int, message string):**

- Clears screen
- Centers and renders message in error style

**main():**

1. Get stdin file descriptor, put terminal in raw mode via `term.MakeRaw()`,
   defer restore
2. Hide cursor, defer show cursor
3. Set up signal handling: ignore SIGINT, SIGTERM, SIGTSTP
4. Handle SIGWINCH to re-render on resize
5. Render lock screen
6. Main loop: a. Read single byte from stdin (blocks until keypress) b. Render
   auth prompt: "Authenticating with Touch ID..." c. Call
   `C.authenticate_touchid()` d. If success: clear screen, restore terminal,
   exit 0 e. If fail: render prompt "Password: " f. Read password input (read
   bytes, print `*` for each char, handle backspace, Enter submits) g. Call
   `C.authenticate_password(C.CString(password))` h. If success: clear screen,
   restore terminal, exit 0 i. If fail: render error "Authentication failed",
   sleep 1s, go to step 5 (re-render lock screen)

```go
package main

/*
#cgo CFLAGS: -x objective-c
#cgo LDFLAGS: -framework LocalAuthentication -framework Foundation -lpam

#include <stdlib.h>
#include <LocalAuthentication/LocalAuthentication.h>
#include <security/pam_appl.h>
#include <pwd.h>
#include <unistd.h>

// Touch ID authentication
int authenticate_touchid() {
    __block int result = 0;
    dispatch_semaphore_t sema = dispatch_semaphore_create(0);

    LAContext *context = [[LAContext alloc] init];
    NSError *error = nil;

    if ([context canEvaluatePolicy:LAPolicyDeviceOwnerAuthenticationWithBiometrics error:&error]) {
        [context evaluatePolicy:LAPolicyDeviceOwnerAuthenticationWithBiometrics
                localizedReason:@"tlock: unlock terminal"
                          reply:^(BOOL success, NSError *evalError) {
            if (success) {
                result = 1;
            }
            dispatch_semaphore_signal(sema);
        }];
        dispatch_semaphore_wait(sema, DISPATCH_TIME_FOREVER);
    }

    return result;
}

// PAM conversation function
static const char *pam_password = NULL;

int pam_conv_func(int num_msg, const struct pam_message **msg,
                  struct pam_response **resp, void *appdata_ptr) {
    struct pam_response *reply = calloc(num_msg, sizeof(struct pam_response));
    if (reply == NULL) return PAM_BUF_ERR;

    for (int i = 0; i < num_msg; i++) {
        if (msg[i]->msg_style == PAM_PROMPT_ECHO_OFF ||
            msg[i]->msg_style == PAM_PROMPT_ECHO_ON) {
            reply[i].resp = strdup(pam_password);
            reply[i].resp_retcode = 0;
        }
    }
    *resp = reply;
    return PAM_SUCCESS;
}

// Password authentication via PAM
int authenticate_password(const char *pw) {
    pam_password = pw;
    struct passwd *pwd = getpwuid(getuid());
    if (pwd == NULL) return 0;

    struct pam_conv conv = { pam_conv_func, NULL };
    pam_handle_t *pamh = NULL;

    int ret = pam_start("login", pwd->pw_name, &conv, &pamh);
    if (ret != PAM_SUCCESS) return 0;

    ret = pam_authenticate(pamh, 0);
    pam_end(pamh, ret);

    return (ret == PAM_SUCCESS) ? 1 : 0;
}
*/
import "C"

import (
	"fmt"
	"os"
	"os/signal"
	"syscall"
	"time"
	"unsafe"

	"github.com/charmbracelet/lipgloss"
	"golang.org/x/term"
)

var (
	purple = lipgloss.Color("99")
	teal   = lipgloss.Color("#06ffa5")
	gray   = lipgloss.Color("245")
	red    = lipgloss.Color("196")

	lockTitleStyle = lipgloss.NewStyle().Bold(true).Foreground(purple)
	subtitleStyle  = lipgloss.NewStyle().Foreground(gray)
	promptStyle    = lipgloss.NewStyle().Foreground(teal)
	errorStyle     = lipgloss.NewStyle().Bold(true).Foreground(red)
)

func clearScreen() {
	fmt.Print("\033[2J\033[H")
}

func hideCursor() {
	fmt.Print("\033[?25l")
}

func showCursor() {
	fmt.Print("\033[?25h")
}

func centerText(text string, width int) string {
	pad := (width - len(text)) / 2
	if pad < 0 {
		pad = 0
	}
	return fmt.Sprintf("%*s%s", pad, "", text)
}

func getTermSize() (int, int) {
	w, h, err := term.GetSize(int(os.Stdout.Fd()))
	if err != nil {
		return 80, 24
	}
	return w, h
}

func renderLockScreen() {
	w, h := getTermSize()
	clearScreen()

	title := lockTitleStyle.Render("LOCKED")
	hostname, _ := os.Hostname()
	host := subtitleStyle.Render(hostname)
	hint := subtitleStyle.Render("Press any key to unlock")

	mid := h / 2
	// Move to vertical center
	fmt.Printf("\033[%d;0H", mid-1)
	fmt.Println(centerText(title, w))
	fmt.Println(centerText(host, w))
	fmt.Println()
	fmt.Println(centerText(hint, w))
}

func renderMessage(msg string, style lipgloss.Style) {
	w, h := getTermSize()
	clearScreen()
	rendered := style.Render(msg)
	fmt.Printf("\033[%d;0H", h/2)
	fmt.Println(centerText(rendered, w))
}

func readPassword() string {
	w, h := getTermSize()
	clearScreen()
	label := promptStyle.Render("Password: ")
	fmt.Printf("\033[%d;0H", h/2)
	fmt.Print(centerText(label, w))
	showCursor()
	defer hideCursor()

	var pw []byte
	buf := make([]byte, 1)
	for {
		n, err := os.Stdin.Read(buf)
		if err != nil || n == 0 {
			continue
		}
		b := buf[0]
		switch {
		case b == 13 || b == 10: // Enter
			fmt.Println()
			return string(pw)
		case b == 127 || b == 8: // Backspace
			if len(pw) > 0 {
				pw = pw[:len(pw)-1]
				fmt.Print("\b \b")
			}
		case b == 3: // Ctrl+C — ignore
			continue
		case b >= 32: // Printable
			pw = append(pw, b)
			fmt.Print("*")
		}
	}
}

func main() {
	fd := int(os.Stdin.Fd())
	oldState, err := term.MakeRaw(fd)
	if err != nil {
		fmt.Fprintf(os.Stderr, "failed to set raw mode: %v\n", err)
		os.Exit(1)
	}
	defer term.Restore(fd, oldState)

	hideCursor()
	defer showCursor()
	defer clearScreen()

	// Ignore signals that could bypass the lock
	signal.Ignore(syscall.SIGINT, syscall.SIGTERM, syscall.SIGTSTP)

	// Handle resize
	sigwinch := make(chan os.Signal, 1)
	signal.Notify(sigwinch, syscall.SIGWINCH)

	for {
		renderLockScreen()

		// Wait for keypress or resize
		buf := make([]byte, 1)
		for {
			// Non-blocking check for resize
			select {
			case <-sigwinch:
				renderLockScreen()
				continue
			default:
			}
			os.Stdin.Read(buf)
			break
		}

		// Try Touch ID
		renderMessage("Authenticating with Touch ID...", promptStyle)
		if C.authenticate_touchid() == 1 {
			return
		}

		// Fall back to password
		pw := readPassword()
		cpw := C.CString(pw)
		defer C.free(unsafe.Pointer(cpw))

		if C.authenticate_password(cpw) == 1 {
			return
		}

		renderMessage("Authentication failed", errorStyle)
		time.Sleep(1 * time.Second)
	}
}
```
